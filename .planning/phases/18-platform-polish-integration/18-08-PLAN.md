```typescript
function buildDateParams(filter: DateRangeFilterValue): string {
  if (filter.preset === "custom" && filter.from && filter.to) {
    return `from=${filter.from}&to=${filter.to}`;
  }
  if (filter.preset !== "custom") {
    return `range=${filter.preset}`;
  }
  return "";
}
```

For client-side filtered exports: filter the data array using the date range before building CSV rows:
```typescript
function filterByDateRange<T extends { createdAt?: string; saleDate?: string }>(
  data: T[],
  filter: DateRangeFilterValue,
): T[] {
  let from: Date | null = null;
  let to: Date | null = null;

  if (filter.preset === "custom" && filter.from && filter.to) {
    from = new Date(filter.from + "T00:00:00Z");
    to = new Date(filter.to + "T23:59:59.999Z");
  } else if (filter.preset === "7d") {
    from = new Date(); from.setDate(from.getDate() - 7); from.setHours(0,0,0,0);
    to = new Date(); to.setHours(23,59,59,999);
  } else if (filter.preset === "30d" || filter.preset === "month") {
    from = new Date(); from.setDate(from.getDate() - 30); from.setHours(0,0,0,0);
    to = new Date(); to.setHours(23,59,59,999);
  }

  if (!from || !to) return data;
  return data.filter(item => {
    const date = new Date(item.saleDate || item.createdAt || "");
    return date >= from! && date <= to!;
  });
}
```

Apply this helper function in each export handler, before building CSV rows.

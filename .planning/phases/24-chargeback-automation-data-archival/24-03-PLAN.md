```typescript
import { prisma } from "@ops/db";
import { randomUUID } from "crypto";
import { logAudit } from "./audit";

const BATCH_SIZE = 5000;

const ARCHIVE_TABLES = [
  { main: "call_audits", archive: "call_audits_archive", dateCol: "created_at" },
  { main: "convoso_call_logs", archive: "convoso_call_logs_archive", dateCol: "created_at" },
  { main: "app_audit_log", archive: "app_audit_log_archive", dateCol: "created_at" },
] as const;

type TableName = typeof ARCHIVE_TABLES[number]["main"];
```

**`previewArchive(cutoffDate: Date)`:**
1. For each table in ARCHIVE_TABLES, query the count of rows eligible for archiving:
   ```sql
   SELECT COUNT(*) as count FROM {main} WHERE {dateCol} < $1
   ```
2. Return `{ tables: { name: string; count: number }[]; total: number }` where `total` is the sum of all counts.

**`archiveRecords(cutoffDate: Date, tables: TableName[], userId: string)`:**
1. Generate a `batchId` via `randomUUID()`.
2. For each table in `ARCHIVE_TABLES` (filtered by `tables` param):
   a. If table is `call_audits`, first null out `call_audit_id` on `convoso_call_logs` for rows that reference call_audits older than cutoffDate:
      ```sql
      UPDATE convoso_call_logs SET call_audit_id = NULL
      WHERE call_audit_id IN (SELECT id FROM call_audits WHERE created_at < $1)
      ```
   b. Count rows: `SELECT COUNT(*) as count FROM {main} WHERE {dateCol} < $1`
   c. If count > 0, batch the archive in chunks of BATCH_SIZE:
      - Loop: `INSERT INTO {archive} SELECT *, NOW(), '{batchId}' FROM {main} WHERE {dateCol} < $1 LIMIT {BATCH_SIZE}`
      - Then: `DELETE FROM {main} WHERE id IN (SELECT id FROM {main} WHERE {dateCol} < $1 LIMIT {BATCH_SIZE})`
      - Repeat until no more rows match.
   d. Record count in results array.
3. Call `logAudit(userId, "data_archived", "Archive", batchId, { tables, cutoffDate: cutoffDate.toISOString(), results })`.
4. Return `{ batchId, results: { table: string; count: number }[] }`.

**`restoreBatch(batchId: string, userId: string)`:**
1. For each table in ARCHIVE_TABLES:
   a. Count: `SELECT COUNT(*) FROM {archive} WHERE archive_batch_id = $1`
   b. If count > 0:
      - Get column names from archive table (exclude `archived_at` and `archive_batch_id`): use the known column list from the main table schema.
      - `INSERT INTO {main} SELECT {columns_without_archived_at_and_batch_id} FROM {archive} WHERE archive_batch_id = $1`
      - `DELETE FROM {archive} WHERE archive_batch_id = $1`
   c. Record count.
2. Call `logAudit(userId, "data_restored", "Archive", batchId, { results })`.
3. Return `{ batchId, results }`.

**`getArchiveStats()`:**
1. For each archive table, query:
   - `SELECT COUNT(*) as count, MIN(created_at) as oldest, MAX(created_at) as newest FROM {archive}`
2. Also query distinct batch IDs: `SELECT DISTINCT archive_batch_id, COUNT(*) as count FROM {archive} GROUP BY archive_batch_id ORDER BY MIN(archived_at) DESC LIMIT 20`
3. Return `{ tables: { name, rowCount, oldestRecord, newestRecord }[], batches: { batchId, table, count, archivedAt }[] }`.

Important: Use `prisma.$queryRawUnsafe` for SELECT queries and `prisma.$executeRawUnsafe` for INSERT/UPDATE/DELETE. Use parameterized queries for dates (`$1`) but table/column names are from the constant array (safe from injection).

```typescript
import { Router } from "express";
import { z } from "zod";
import { requireAuth, requireRole } from "../middleware/auth";
import { archiveRecords, restoreBatch, getArchiveStats, previewArchive } from "../services/archive";
import { zodErr, asyncHandler } from "./helpers";

const router = Router();

const archiveSchema = z.object({
  cutoffDays: z.number().int().min(1).default(90),
  tables: z.array(z.enum(["call_audits", "convoso_call_logs", "app_audit_log"])).min(1),
});

// GET /archive/preview -- return eligible row counts without archiving
router.get("/archive/preview", requireAuth, requireRole("SUPER_ADMIN", "OWNER_VIEW"), asyncHandler(async (req, res) => {
  const days = parseInt(String(req.query.cutoffDays ?? "90"), 10);
  if (isNaN(days) || days < 1) return res.status(400).json({ error: "cutoffDays must be a positive integer" });

  const cutoffDate = new Date();
  cutoffDate.setDate(cutoffDate.getDate() - days);

  const result = await previewArchive(cutoffDate);
  return res.json(result);
}));

// POST /archive -- archive records older than cutoff
router.post("/archive", requireAuth, requireRole("SUPER_ADMIN", "OWNER_VIEW"), asyncHandler(async (req, res) => {
  const parsed = archiveSchema.safeParse(req.body);
  if (!parsed.success) return res.status(400).json(zodErr(parsed.error));

  const cutoffDate = new Date();
  cutoffDate.setDate(cutoffDate.getDate() - parsed.data.cutoffDays);

  const result = await archiveRecords(cutoffDate, parsed.data.tables, req.user!.id);
  return res.json(result);
}));

const restoreSchema = z.object({
  batchId: z.string().min(1),
});

// POST /archive/restore -- restore an archived batch
router.post("/archive/restore", requireAuth, requireRole("SUPER_ADMIN", "OWNER_VIEW"), asyncHandler(async (req, res) => {
  const parsed = restoreSchema.safeParse(req.body);
  if (!parsed.success) return res.status(400).json(zodErr(parsed.error));

  const result = await restoreBatch(parsed.data.batchId, req.user!.id);
  return res.json(result);
}));

// GET /archive/stats -- get archive statistics
router.get("/archive/stats", requireAuth, requireRole("SUPER_ADMIN", "OWNER_VIEW"), asyncHandler(async (_req, res) => {
  const stats = await getArchiveStats();
  return res.json(stats);
}));

export default router;
```

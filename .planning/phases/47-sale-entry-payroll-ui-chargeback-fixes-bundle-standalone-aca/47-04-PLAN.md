Extract the ACA covering child sale creation logic from POST /sales/aca into a transaction-aware helper:

```ts
import type { Prisma } from "@prisma/client";

export async function createAcaChildSale(
  tx: Prisma.TransactionClient,
  parentSaleId: string,
  input: { productId: string; memberCount: number; userId: string },
): Promise<{ id: string }> {
  // 1. Fetch parent sale for agent, memberName, saleDate
  const parent = await tx.sale.findUnique({
    where: { id: parentSaleId },
    select: { id: true, agentId: true, memberName: true, memberId: true, saleDate: true, leadSourceId: true },
  });
  if (!parent) throw new Error(`Parent sale ${parentSaleId} not found`);

  // 2. Fetch ACA_PL product for acaRate
  const product = await tx.product.findUnique({ where: { id: input.productId } });
  if (!product || product.type !== "ACA_PL") {
    throw new Error(`Product ${input.productId} is not ACA_PL`);
  }
  const acaRate = Number(product.acaRate ?? 0);
  const payout = acaRate * input.memberCount;

  // 3. Create the child sale with acaCoveringSaleId pointing UP at parent
  //    (per research drift correction — the FK is on the child, not parent)
  const child = await tx.sale.create({
    data: {
      agentId: parent.agentId,
      productId: input.productId,
      memberName: parent.memberName,
      memberId: parent.memberId,
      saleDate: parent.saleDate,
      leadSourceId: parent.leadSourceId,
      memberCount: input.memberCount,
      premium: 0, // standalone ACA convention — premium always 0, payout is acaRate × memberCount
      status: "approved",
      acaCoveringSaleId: parent.id, // FK points UP at parent
    },
    select: { id: true },
  });

  return child;
}

export async function removeAcaChildSale(
  tx: Prisma.TransactionClient,
  parentSaleId: string,
  childSaleId: string,
): Promise<void> {
  // Delete child's PayrollEntry first (FK constraint)
  await tx.payrollEntry.deleteMany({ where: { saleId: childSaleId } });
  // Delete child Sale — acaCoveringSaleId link is removed naturally
  await tx.sale.delete({ where: { id: childSaleId } });
}
```

**4. Add `handleSaleEditApproval` to payroll.ts:**

Export a new function after `upsertPayrollEntryForSale`:

```typescript
export const handleSaleEditApproval = async (saleId: string, changes: Record<string, { old: any; new: any }>, oldAgentId?: string) => {
  // If agent changed, delete old payroll entries under old agent
  if (oldAgentId && changes.agentId) {
    await prisma.payrollEntry.deleteMany({ where: { saleId, agentId: oldAgentId } });
  }

  // Check if the sale's current payroll entry is in a finalized period
  const existingEntries = await prisma.payrollEntry.findMany({
    where: { saleId },
    include: { payrollPeriod: true },
  });

  const finalizedEntry = existingEntries.find(e =>
    e.payrollPeriod.status === 'FINALIZED' || e.payrollPeriod.status === 'LOCKED'
  );

  if (finalizedEntry) {
    // Calculate old commission from the finalized entry
    const oldPayout = Number(finalizedEntry.payoutAmount);

    // Recalculate with new values to get new commission
    await upsertPayrollEntryForSale(saleId);

    // The upsert may have created a new entry in a different period (if date/payment changed)
    // Find the new entry
    const newEntries = await prisma.payrollEntry.findMany({ where: { saleId } });
    const newEntry = newEntries.find(e => e.id !== finalizedEntry.id) || newEntries[0];

    if (newEntry && newEntry.id !== finalizedEntry.id) {
      // Different period — new entry already correct, mark old as CLAWBACK_APPLIED
      await prisma.payrollEntry.update({
        where: { id: finalizedEntry.id },
        data: {
          adjustmentAmount: Number(finalizedEntry.adjustmentAmount) - oldPayout,
          status: 'CLAWBACK_APPLIED',
        },
      });
    } else {
      // Same period but finalized — create adjustment in next open period
      // Zero out the finalized entry
      await prisma.payrollEntry.update({
        where: { id: finalizedEntry.id },
        data: {
          adjustmentAmount: Number(finalizedEntry.adjustmentAmount) - oldPayout,
          status: 'CLAWBACK_APPLIED',
        },
      });
      // upsertPayrollEntryForSale already created new entry in correct period
    }
  } else {
    // No finalized period — just recalculate normally
    await upsertPayrollEntryForSale(saleId);
  }
};
```

**1. Extend PATCH /api/sales/:id (currently at line 340):**

Replace the current restricted PATCH with a role-aware version:

- Accept ALL sale fields: `saleDate`, `agentId`, `memberName`, `memberId`, `carrier`, `productId`, `premium`, `effectiveDate`, `leadSourceId`, `enrollmentFee`, `addonProductIds`, `addonPremiums`, `paymentType`, `memberState`, `notes`, `commissionApproved`
- Open to MANAGER, PAYROLL, SUPER_ADMIN (add MANAGER to requireRole)
- Role branching:
  - **PAYROLL / SUPER_ADMIN**: Apply changes directly. If `productId` or `addonProductIds` changed, delete old SaleAddons and create new ones in a transaction. If `agentId` changed, delete old payroll entries for old agent. Call `upsertPayrollEntryForSale(saleId)` if any financial/product/agent/date/paymentType field changed. Return updated sale.
  - **MANAGER**: Do NOT apply changes. Instead:
    1. Check for pending StatusChangeRequest on this sale — if exists, return 409 `{ error: "This sale has a pending status change. Wait for resolution before editing." }`
    2. Check for pending SaleEditRequest on this sale — if exists, return 409 `{ error: "This sale already has a pending edit request." }`
    3. Fetch current sale with product + addons to build the `changes` diff JSON
    4. Build diff: `{ fieldName: { old: currentValue, new: newValue } }` for each changed field
    5. Call preview endpoint logic to get old and new commission amounts for the diff
    6. Create `SaleEditRequest` with `{ saleId, requestedBy: req.user.id, changes: diffJson }`
    7. Return `{ editRequest, message: "Edit request created for payroll approval" }`

router.get("/sales/:id", requireAuth, asyncHandler(async (req, res) => {
  const sale = await prisma.sale.findUnique({
    where: { id: req.params.id },
    include: {
      agent: true, product: true, leadSource: true,
      addons: { include: { product: true } },
      _count: {
        select: {
          statusChangeRequests: { where: { status: 'PENDING' } },
          saleEditRequests: { where: { status: 'PENDING' } },
        },
      },
    },
  });
  if (!sale) return res.status(404).json({ error: "Sale not found" });
  const { _count, ...rest } = sale;
  res.json({
    ...rest,
    hasPendingStatusChange: _count.statusChangeRequests > 0,
    hasPendingEditRequest: _count.saleEditRequests > 0,
  });
}));

   ```typescript
   if (range === "last_week") {
     const day = now.getDay(); // 0=Sun ... 6=Sat
     const thisSunday = new Date(todayStart);
     thisSunday.setDate(todayStart.getDate() - day);
     const lastSunday = new Date(thisSunday);
     lastSunday.setDate(thisSunday.getDate() - 7);
   ```
